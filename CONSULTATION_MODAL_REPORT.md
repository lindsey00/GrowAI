```typescript

  setFormData({ ...formData, [e.target.name]: e.target.value });
};

// 관심 분야 토글
const handleConcernToggle = (concernId) => {
  setFormData({
    ...formData,
    concerns: formData.concerns.includes(concernId)
      ? formData.concerns.filter((c) => c !== concernId)
      : [...formData.concerns, concernId],
  });
};
```
