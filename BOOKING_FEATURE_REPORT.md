const isDateAvailable = (date: Date | null) => {
  if (!date) return false;
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  const dayOfWeek = date.getDay();
  return date >= today && dayOfWeek !== 0 && dayOfWeek !== 6;
};
